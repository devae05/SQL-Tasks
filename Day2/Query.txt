create database day1;
use day1;
set sql_safe_updates=0;

-- student
desc student;
create table Student(rollno int,name varchar(50),std varchar(5));
insert into Student values(101,"Kisore",'7th');
insert into Student values(102,"Ram",'9th');
insert into Student values(103,"Kenny",'8th');
insert into Student values(104,"Joe",'12th');
alter table student modify name varchar(20);
alter table student add age int;
alter table student drop city;
delete from student where rollno=104; 
update student set age=18 where rollno=103;
select * from Student;
